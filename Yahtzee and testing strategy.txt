import random

# Function to roll the dice
def roll_dice(num_dice):
    return [random.randint(1, 6) for _ in range(num_dice)]

# Function to calculate score for upper section categories
def calculate_upper_section_score(dice, number):
    return dice.count(number) * number

# Function to check if a list is a consecutive sequence
def is_consecutive(lst):
    return sorted(lst) in [[1, 2, 3, 4, 5], [2, 3, 4, 5, 6]]

# Function to calculate score for lower section categories
def calculate_lower_section_score(dice, category):
    if category == "Three of a Kind":
        return sum(dice) if len(set(dice)) <= 3 else 0
    elif category == "Four of a Kind":
        return sum(dice) if len(set(dice)) <= 2 else 0
    elif category == "Full House":
        return 25 if len(set(dice)) == 2 and (dice.count(dice[0]) == 2 or dice.count(dice[0]) == 3) else 0
    elif category == "Small Straight":
        return 30 if is_consecutive(sorted(dice)[:4]) else 0
    elif category == "Large Straight":
        return 40 if is_consecutive(dice) else 0
    elif category == "Yahtzee":
        return 50 if len(set(dice)) == 1 else 0
    elif category == "Chance":
        return sum(dice)
    else:
        return 0

# Function to validate category choice
def validate_category(category, dice, scorecard):
    if category in scorecard:
        return False, "Category already scored!"
    if category in ["Ones", "Twos", "Threes", "Fours", "Fives", "Sixes"]:
        if category not in dice:
            return False, f"No {category} in the roll!"
    elif category == "Three of a Kind":
        if len(set(dice)) < 3:
            return False, "Requires at least three of the same number!"
    elif category == "Four of a Kind":
        if len(set(dice)) < 2:
            return False, "Requires at least four of the same number!"
    elif category == "Full House":
        if len(set(dice)) != 2 or (dice.count(dice[0]) != 2 and dice.count(dice[0]) != 3):
            return False, "Requires three of one number and two of another!"
    elif category == "Small Straight":
        if not is_consecutive(sorted(dice)[:4]):
            return False, "Requires a sequence of four numbers!"
    elif category == "Large Straight":
        if not is_consecutive(dice):
            return False, "Requires a sequence of five numbers!"
    elif category == "Yahtzee":
        if len(set(dice)) != 1:
            return False, "Requires all dice to have the same number!"
    return True, ""
